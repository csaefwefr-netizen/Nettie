        // --- 狀態變量 ---
        let lang = 'tc'; 
        let studentName = '';
        let currentStep = 'login'; // login, match, mcq, done
        
        // Match 狀態
        let selectedLeftId = null;
        let matchedPairs =[];

        // MCQ 狀態
        let currentMcqIndex = 0;

        // --- 核心函數 ---

        function getDict() {
            return data[lang];
        }

        function toggleLanguage() {
            lang = lang === 'tc' ? 'sc' : 'tc';
            document.getElementById('lang-btn').innerText = lang === 'tc' ? '繁 / 简' : '简 / 繁';
            document.getElementById('app-title').innerText = getDict().title;
            render();
        }

        function showToast(msg, isSuccess = false) {
            const toast = document.getElementById('toast');
            toast.innerText = msg;
            if (isSuccess) {
                toast.classList.add('success');
            } else {
                toast.classList.remove('success');
            }
            toast.classList.add('show');
            setTimeout(() => {
                toast.classList.remove('show');
            }, 2500);
        }

        // --- 渲染邏輯 ---

        function render() {
            const content = document.getElementById('content-area');
            const dict = getDict();
            
            if (currentStep === 'login') {
                content.innerHTML = `
                    <div class="login-screen">
                        <h2>${dict.namePrompt}</h2>
                        <input type="text" id="student-name-input" placeholder="...">
                        <br>
                        <button class="btn" onclick="startExercise()">${dict.startBtn}</button>
                    </div>
                `;
                // 保留已經輸入的名字
                if(studentName) document.getElementById('student-name-input').value = studentName;
            } 
            else if (currentStep === 'match') {
                let leftHtml = '';
                let rightHtml = '';

                dict.matchLeft.forEach(item => {
                    const isMatched = matchedPairs.includes(item.id);
                    const isSelected = selectedLeftId === item.id;
                    let classes = 'match-item';
                    if (isMatched) classes += ' matched';
                    else if (isSelected) classes += ' selected';

                    leftHtml += `<div class="${classes}" onclick="selectLeft(${item.id})">${item.text}</div>`;
                });

                dict.matchRight.forEach(item => {
                    const isMatched = matchedPairs.includes(item.id);
                    let classes = 'match-item';
                    if (isMatched) classes += ' matched';

                    rightHtml += `<div class="${classes}" onclick="selectRight(${item.id})">${item.text}</div>`;
                });

                content.innerHTML = `
                    <div class="student-name-display">${dict.studentLabel} ${studentName}</div>
                    <div class="section-title">${dict.part1Title}</div>
                    <div class="section-desc">${dict.part1Desc}</div>
                    <div class="match-container">
                        <div class="match-col">${leftHtml}</div>
                        <div class="match-col">${rightHtml}</div>
                    </div>
                `;
            }
            else if (currentStep === 'mcq') {
                const qData = dict.mcq[currentMcqIndex];
                let optionsHtml = '';
                qData.options.forEach((opt, index) => {
                    optionsHtml += `<div class="mcq-option" onclick="selectMcq(${index})">${opt}</div>`;
                });

                content.innerHTML = `
                    <div class="student-name-display">${dict.studentLabel} ${studentName}</div>
                    <div class="section-title">${dict.part2Title}</div>
                    <div class="section-desc">進度： ${currentMcqIndex + 1} / ${dict.mcq.length}</div>
                    <div class="mcq-question">${qData.q}</div>
                    <div class="mcq-options">${optionsHtml}</div>
                `;
            }
            else if (currentStep === 'done') {
                content.innerHTML = `
                    <div class="login-screen">
                        <h2 style="color: var(--success);">${dict.finishTitle}</h2>
                        <p>${studentName} ${dict.studentLabel.replace('：','')}，${dict.finishDesc}</p>
                    </div>
                `;
            }
        }

        // --- 交互邏輯 ---

        function startExercise() {
            const input = document.getElementById('student-name-input').value.trim();
            if (!input) {
                showToast(getDict().nameAlert);
                return;
            }
            studentName = input;
            currentStep = 'match';
            render();
        }

        function selectLeft(id) {
            if (matchedPairs.includes(id)) return;
            selectedLeftId = id;
            render();
        }

        function selectRight(rightId) {
            if (matchedPairs.includes(rightId)) return;
            
            if (selectedLeftId === null) {
                return; // 未選擇左側，無反應
            }

            const dict = getDict();
            if (selectedLeftId === rightId) {
                // 配對成功
                matchedPairs.push(rightId);
                selectedLeftId = null;
                showToast(dict.correctMsg, true);
                render();

                // 檢查是否全對
                if (matchedPairs.length === dict.matchLeft.length) {
                    setTimeout(() => {
                        showToast(dict.matchNextMsg, true);
                        setTimeout(() => {
                            currentStep = 'mcq';
                            render();
                        }, 1500);
                    }, 500);
                }
            } else {
                // 配對失敗，找出左側選項的關鍵詞提示
                const leftItem = dict.matchLeft.find(item => item.id === selectedLeftId);
                showToast(`❌ ${dict.hintPrefix}（${leftItem.hint}）`);
                selectedLeftId = null;
                render();
            }
        }

        function selectMcq(selectedIndex) {
            const dict = getDict();
            const qData = dict.mcq[currentMcqIndex];

            if (selectedIndex === qData.ans) {
                // 答對
                showToast(dict.correctMsg, true);
                setTimeout(() => {
                    currentMcqIndex++;
                    if (currentMcqIndex >= dict.mcq.length) {
                        currentStep = 'done';
                    }
                    render();
                }, 1000);
            } else {
                // 答錯，給出關鍵詞提示
                showToast(`❌ ${dict.hintPrefix}（${qData.hint}）`);
            }
        }

        // 初始化
        document.getElementById('app-title').innerText = getDict().title;
        render();
